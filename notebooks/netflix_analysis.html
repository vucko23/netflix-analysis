## 📊 Movies vs TV Shows

netflix %>%
  count(type) %>%
  ggplot(aes(x = type, y = n, fill = type)) +
  geom_bar(stat = "identity") +
  labs(title = "Movies vs TV Shows on Netflix",
       x = "Type", y = "Count") +
  theme_minimal()

## 📈 Releases over time

netflix %>%
  group_by(release_year) %>%
  summarise(count = n()) %>%
  ggplot(aes(x = release_year, y = count)) +
  geom_line(color = "red") +
  labs(title = "Number of Releases Over Time",
       x = "Release Year", y = "Count") +
  theme_minimal()

## 🎭 Top Genres

# Split genres (listed_in column)
genres <- netflix %>%
  separate_rows(listed_in, sep = ", ") %>%
  count(listed_in, sort = TRUE)

## 🌍 Top Countries

netflix %>%
  filter(!is.na(country)) %>%
  count(country, sort = TRUE) %>%
  slice_max(n, n = 10) %>%
  ggplot(aes(x = reorder(country, n), y = n)) +
  geom_bar(stat = "identity", fill = "darkgreen") +
  coord_flip() +
  labs(title = "Top 10 Countries on Netflix",
       x = "Country", y = "Count") +
  theme_minimal()
